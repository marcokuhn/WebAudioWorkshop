What is NexusUI?

NexusUI is a JavaScript library providing touch-optimized UI components specifically designed for web audio applications. It includes dials, sliders, buttons, keyboards, and more - all optimized for both desktop and mobile devices.

Key Features: Touch & mouse support, customizable appearance, easy event handling, and seamless integration with Tone.js and Web Audio API.

Piano Keyboard

Play a note!
// Piano Keyboard var keyboard = new Nexus.Piano('#keyboard1', { size: [600, 120], mode: 'button', // 'button' or 'toggle' lowNote: 48, // C3 highNote: 72 // C5 }); keyboard.on('change', function(v) { if (v.state) { console.log('Note On:', v.note); } else { console.log('Note Off:', v.note); } });

Sequencer Matrix

Click cells to create patterns
// Step Sequencer var sequencer = new Nexus.Sequencer('#sequencer1', { size: [400, 200], mode: 'toggle', rows: 4, // Number of tracks/notes columns: 8 // Number of steps }); sequencer.on('change', function(v) { console.log('Step:', v.column, 'Note:', v.row, 'State:', v.state); });

üéµ Integration with Tone.js

See how to connect NexusUI controls to Tone.js audio parameters.

Note:

This is a basic NexusUI demonstration. For a complete audio integration example, see synth-interface.html

// Complete Tone.js integration example // Create Tone.js synthesizer const synth = new Tone.Synth().toDestination(); // Frequency Dial var freqDial = new Nexus.Dial('#freqDial'); freqDial.on('change', function(value) { var frequency = value * 1000 + 20; // 20-1020 Hz synth.frequency.rampTo(frequency, 0.1); }); // Volume Slider var volumeSlider = new Nexus.Slider('#volumeSlider'); volumeSlider.on('change', function(value) { synth.volume.rampTo(Tone.gainToDb(value), 0.1); }); // Play Button var playBtn = new Nexus.Button('#playBtn'); playBtn.on('change', function(state) { if (state) { synth.triggerAttack('C4'); } else { synth.triggerRelease(); } }); // Piano Keyboard var piano = new Nexus.Piano('#piano'); piano.on('change', function(v) { if (v.state) { synth.triggerAttack(Tone.Frequency(v.note, 'midi')); } else { synth.triggerRelease(); } });
